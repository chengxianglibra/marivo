```python
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Callable, Literal

ContractStatus = Literal["passed", "failed", "xfail", "skipped"]


@dataclass(frozen=True)
class ContractCase:
    name: str
    run: Callable[[Any, Path], Any]


@dataclass(frozen=True)
class ContractResult:
    adapter_name: str
    case_name: str
    status: ContractStatus
    detail: str = ""


def run_contract_cases(
    *,
    adapter_name: str,
    factory: Callable[[Path], Any],
    cases: list[ContractCase],
    tmp_path: Path,
) -> list[ContractResult]:
    results: list[ContractResult] = []
    adapter = factory(tmp_path)
    for case in cases:
        try:
            case.run(adapter, tmp_path)
        except Exception as exc:  # contract harness reports raw failure detail
            results.append(ContractResult(adapter_name, case.name, "failed", str(exc)))
        else:
            results.append(ContractResult(adapter_name, case.name, "passed"))
    return results
```

```python
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Callable


@dataclass(frozen=True)
class ParityResult:
    case_name: str
    local_status: str
    remote_status: str
    detail: str = ""


def compare_contract_matrix(
    *,
    local_name: str,
    local_factory: Callable[[Path], Any],
    remote_name: str,
    remote_factory: Callable[[Path], Any],
    cases: list[Any],
    tmp_path: Path,
) -> list[ParityResult]:
    results: list[ParityResult] = []
    local_adapter = local_factory(tmp_path)
    remote_adapter = remote_factory(tmp_path)
    for case in cases:
        try:
            case.run(local_adapter, tmp_path)
            local_status = "passed"
        except Exception as exc:
            local_status = f"failed:{exc}"
        try:
            case.run(remote_adapter, tmp_path)
            remote_status = "passed"
        except Exception as exc:
            remote_status = f"failed:{exc}"
        results.append(
            ParityResult(
                case_name=case.name,
                local_status=local_status,
                remote_status=remote_status,
            )
        )
    return results
```
